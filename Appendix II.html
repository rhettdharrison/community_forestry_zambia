Appendix 1 - CFs Surveyed

Community Forest Management in Zambia: Is It Working?

List of CFs surveyed and number of interviews

Code 
cfmg |>
  group_by(province, cfmg_name) |>
  summarise(n_interviews = n(), .groups = "drop") |>
  mutate(cfmg_name = str_to_sentence(cfmg_name)) |>
  arrange(province, cfmg_name) |>
  gt(groupname_col = "province") |>
  cols_label(
    cfmg_name    = "Community forest",
    n_interviews = "Interviews"
  ) |>
  cols_align(align = "left",   columns = cfmg_name) |>
  cols_align(align = "center", columns = n_interviews) |>
  summary_rows(
    groups  = everything(),
    columns = n_interviews,
    fns     = list(Total = ~sum(.)),
    fmt     = ~fmt_integer(.)
  )
